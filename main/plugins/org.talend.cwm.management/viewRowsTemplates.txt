SqlTemplate.SQL.UDI.Match=SELECT COUNT(CASE WHEN <MATCHING_EXPRESSION_TEXT_FIELD> THEN 1 END), COUNT(*) FROM  <%=__TABLE_NAME__%> WHERE <WHERE_TEXT_FIELD> <%=__AND_WHERE_CLAUSE__%>

SqlTemplate.SQL.UDI.Match.ViewInvalidValues=SELECT  <%=__COLUMN_NAMES__%> FROM  <%=__TABLE_NAME__%> WHERE NOT (<MATCHING_EXPRESSION_TEXT_FIELD> AND <WHERE_TEXT_FIELD>) <%=__AND_WHERE_CLAUSE__%>
SqlTemplate.SQL.UDI.Match.ViewValidRows=SELECT * FROM  <%=__TABLE_NAME__%> WHERE <MATCHING_EXPRESSION_TEXT_FIELD> AND <WHERE_TEXT_FIELD> <%=__AND_WHERE_CLAUSE__%>
SqlTemplate.SQL.UDI.Match.ViewInvalidRows=SELECT * FROM  <%=__TABLE_NAME__%> WHERE NOT (<MATCHING_EXPRESSION_TEXT_FIELD> AND <WHERE_TEXT_FIELD>) <%=__AND_WHERE_CLAUSE__%>
